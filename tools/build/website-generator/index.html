function switchWsTab(tab) {
  const frame = document.getElementById('wsPreviewFrame');
  const code = document.getElementById('wsCodeView');
  if (tab === 'preview') {
    frame.style.display = 'block'; code.style.display = 'none';
    document.getElementById('wsCodeToggle').style.display = '';
    document.getElementById('wsPreviewToggle').style.display = 'none';
  } else {
    frame.style.display = 'none'; code.style.display = 'block';
    document.getElementById('wsCodeToggle').style.display = 'none';
    document.getElementById('wsPreviewToggle').style.display = '';
  }
}
