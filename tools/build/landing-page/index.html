function switchLpTab(tab) {
  document.getElementById('lpCodeView').style.display = tab === 'code' ? '' : 'none';
  document.getElementById('lpPreviewView').style.display = tab === 'preview' ? '' : 'none';
  document.getElementById('lpCodeTab').classList.toggle('active', tab === 'code');
  document.getElementById('lpPreviewTab').classList.toggle('active', tab === 'preview');
}
